import React, { useState, useEffect } from 'react';
import { Calculator, Heart, Brain, DollarSign, Calendar, TrendingUp,
AlertCircle, CheckCircle, Target, Users } from 'lucide-react';
const SmokingCessationCalculator = () => {
const [formData, setFormData] = useState({
cigarettesPerDay: '',
yearsSmoked: '',
pricePerPack: '',
cigarettesPerPack: 20,
quitDate: new Date().toISOString().split('T')[0],
age: '',
gender: '',
location: 'US',
healthConditions: [],
smokingTriggers: [],
supportSystem: 3,
motivation: 3,
previousAttempts: 0,
stressLevel: 3
});
const [results, setResults] = useState(null);
const [eqScore, setEqScore] = useState(null);
const [activeTab, setActiveTab] = useState('calculator');
const healthConditions = [
'Heart Disease', 'High Blood Pressure', 'Diabetes', 'Asthma',
'COPD', 'Depression', 'Anxiety', 'Pregnancy', 'None'
];
const triggers = [
'Stress', 'Social Situations', 'Alcohol', 'Coffee/Meals',
'Driving', 'Work Breaks', 'Boredom', 'Anger', 'Celebration'
];
const eqQuestions = [
{
id: 'selfAware',
question: 'How well do you understand your emotional triggers for
smoking?',
options: ['Not at all', 'Somewhat', 'Moderately', 'Very well',
'Extremely well']
},
{
id: 'stressManage',
question: 'How confident are you in managing stress without
smoking?',
options: ['Not confident', 'Slightly confident', 'Moderately
confident', 'Very confident', 'Extremely confident']
},
{
id: 'socialPressure',
question: 'How well can you resist peer pressure to smoke?',
options: ['Very poorly', 'Poorly', 'Moderately', 'Well', 'Very
well']
},
{
id: 'emotionalControl',
question: 'How well do you control your emotions during difficult
times?',
options: ['Very poorly', 'Poorly', 'Moderately', 'Well', 'Very
well']
},
{
id: 'motivation',
question: 'How motivated are you to quit smoking permanently?',
options: ['Not motivated', 'Slightly motivated', 'Moderately
motivated', 'Very motivated', 'Extremely motivated']
}
];
const [eqAnswers, setEqAnswers] = useState({});
const calculateResults = () => {
const cigarettesPerDay = parseFloat(formData.cigarettesPerDay);
const yearsSmoked = parseFloat(formData.yearsSmoked);
const pricePerPack = parseFloat(formData.pricePerPack);
const cigarettesPerPack = parseFloat(formData.cigarettesPerPack);
if (!cigarettesPerDay || !yearsSmoked || !pricePerPack) return;
const quitDate = new Date(formData.quitDate);
const today = new Date();
const daysSinceQuit = Math.max(0, Math.floor((today - quitDate) /
(1000 * 60 * 60 * 24)));
// Financial calculations
const dailyCost = (cigarettesPerDay / cigarettesPerPack) *
pricePerPack;
const weeklySavings = dailyCost * 7;
const monthlySavings = dailyCost * 30;
const yearlySavings = dailyCost * 365;
const totalSaved = dailyCost * daysSinceQuit;
// Health calculations
const totalCigarettes = cigarettesPerDay * yearsSmoked * 365;
const cigarettesAvoided = cigarettesPerDay * daysSinceQuit;
const lifeMinutesRegained = cigarettesAvoided * 14; // 14 minutes per
cigarette
const hoursRegained = Math.floor(lifeMinutesRegained / 60);
const daysRegained = Math.floor(hoursRegained / 24);
// Health milestones
const healthMilestones = [
{ time: '20 minutes', benefit: 'Heart rate and blood pressure drop'
},
{ time: '12 hours', benefit: 'Carbon monoxide level in blood drops
to normal' },
{ time: '2 weeks - 3 months', benefit: 'Circulation improves and
lung function increases' },
{ time: '1 - 12 months', benefit: 'Coughing and shortness of breath
decrease' },
{ time: '1 year', benefit: 'Risk of coronary heart disease is half
that of a smoker' },
{ time: '5 years', benefit: 'Risk of stroke reduced to that of a
non-smoker' },
{ time: '10 years', benefit: 'Risk of lung cancer falls to half
that of a smoker' },
{ time: '15 years', benefit: 'Risk of coronary heart disease equals
that of a non-smoker' }
];
// Risk assessment based on personal factors
const riskFactors = calculateRiskFactors();
setResults({
daysSinceQuit,
totalSaved,
weeklySavings,
monthlySavings,
yearlySavings,
cigarettesAvoided,
lifeMinutesRegained,
hoursRegained,
daysRegained,
healthMilestones,
riskFactors,
totalCigarettes
});
};
const calculateRiskFactors = () => {
let riskScore = 0;
const factors = [];
// Age factor
const age = parseInt(formData.age);
if (age > 50) {
riskScore += 2;
factors.push('Age increases health risks from continued smoking');
}
// Health conditions
if (formData.healthConditions.length > 0 &&
!formData.healthConditions.includes('None')) {
riskScore += formData.healthConditions.length;
factors.push(`Existing health conditions:
${formData.healthConditions.join(', ')}`);
}
// Years smoked
const yearsSmoked = parseFloat(formData.yearsSmoked);
if (yearsSmoked > 20) {
riskScore += 3;
factors.push('Long-term smoking history increases urgency to
quit');
}
// Daily consumption
const cigarettesPerDay = parseFloat(formData.cigarettesPerDay);
if (cigarettesPerDay > 20) {
riskScore += 2;
factors.push('Heavy smoking pattern');
}
return { score: riskScore, factors };
};
const calculateEQ = () => {
const answers = Object.values(eqAnswers);
if (answers.length < eqQuestions.length) return;
const totalScore = answers.reduce((sum, answer) => sum + answer, 0);
const maxScore = eqQuestions.length * 4;
const eqPercentage = (totalScore / maxScore) * 100;
let interpretation = '';
let recommendations = [];
if (eqPercentage < 40) {
interpretation = 'Lower Emotional Readiness - High Support Needed';
recommendations = [
'Consider professional counseling before quitting',
'Build a strong support network',
'Start with stress management techniques',
'Consider gradual reduction approach',
'Focus on emotional coping skills development'
];
} else if (eqPercentage < 60) {
interpretation = 'Moderate Emotional Readiness - Structured Support
Beneficial';
recommendations = [
'Develop specific trigger management plans',
'Practice stress reduction techniques daily',
'Join a support group or find a quit buddy',
'Consider nicotine replacement therapy',
'Create detailed emergency coping plans'
];
} else if (eqPercentage < 80) {
interpretation = 'Good Emotional Readiness - Self-Directed Success
Likely';
recommendations = [
'Continue developing emotional awareness',
'Create comprehensive quit plan',
'Build support network but rely on internal motivation',
'Consider helping others as additional motivation',
'Focus on long-term maintenance strategies'
];
} else {
interpretation = 'High Emotional Readiness - Excellent Success
Potential';
recommendations = [
'You have strong emotional foundations for success',
'Consider becoming a mentor for others',
'Focus on challenging situations to build resilience',
'Create ambitious health and wellness goals',
'Document your journey to help others'
];
}
setEqScore({
score: Math.round(eqPercentage),
interpretation,
recommendations
});
};
const handleInputChange = (field, value) => {
setFormData(prev => ({
...prev,
[field]: value
}));
};
const handleArrayToggle = (field, value) => {
setFormData(prev => ({
...prev,
[field]: prev[field].includes(value)
? prev[field].filter(item => item !== value)
: [...prev[field], value]
}));
};
const handleEQAnswer = (questionId, answerIndex) => {
setEqAnswers(prev => ({
...prev,
[questionId]: answerIndex
}));
};
useEffect(() => {
if (Object.keys(eqAnswers).length === eqQuestions.length) {
calculateEQ();
}
}, [eqAnswers]);
const TabButton = ({ id, label, icon: Icon }) => (
<button
onClick={() => setActiveTab(id)}
className={`flex items-center px-4 py-2 font-medium rounded-lg
transition-colors ${
activeTab === id
? 'bg-blue-600 text-white'
: 'text-gray-600 hover:text-blue-600 hover:bg-blue-50'
}`}
>
<Icon className="w-4 h-4 mr-2" />
{label}
</button>
);
return (
<div className="max-w-6xl mx-auto p-6 bg-gradient-to-br from-blue-50
to-green-50 min-h-screen">
<div className="bg-white rounded-2xl shadow-2xl overflow-hidden">
{/* Header */}
<div className="bg-gradient-to-r from-blue-600 to-green-600 text-
white p-8">
<div className="flex items-center mb-4">
<Heart className="w-8 h-8 mr-3" />
<h1 className="text-3xl font-bold">Complete Smoking Cessation
Calculator</h1>
</div>
<p className="text-blue-100 text-lg">
Advanced calculator with emotional readiness assessment,
personalized insights, and comprehensive tracking
</p>
</div>
{/* Navigation Tabs */}
<div className="border-b border-gray-200 bg-gray-50 px-8 py-4">
<div className="flex space-x-2 overflow-x-auto">
<TabButton id="calculator" label="Calculator"
icon={Calculator} />
<TabButton id="eq-assessment" label="EQ Assessment"
icon={Brain} />
<TabButton id="results" label="Results" icon={TrendingUp} />
<TabButton id="insights" label="Personal Insights"
icon={Target} />
</div>
</div>
<div className="p-8">
{activeTab === 'calculator' && (
<div className="space-y-8">
<h2 className="text-2xl font-bold text-gray-800 mb-
6">Personal Smoking Profile</h2>
{/* Basic Information */}
<div className="grid md:grid-cols-2 gap-6">
<div className="space-y-4">
<h3 className="text-lg font-semibold text-gray-
700">Smoking Habits</h3>
<div>
<label className="block text-sm font-medium text-
gray-700 mb-2">
Cigarettes per day
</label>
<input
type="number"
value={formData.cigarettesPerDay}
onChange={(e) =>
handleInputChange('cigarettesPerDay', e.target.value)}
className="w-full px-4 py-2 border border-gray-300
rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
placeholder="e.g., 20"
/>
</div>
<div>
<label className="block text-sm font-medium text-
gray-700 mb-2">
Years you've been smoking
</label>
<input
type="number"
value={formData.yearsSmoked}
onChange={(e) => handleInputChange('yearsSmoked',
e.target.value)}
className="w-full px-4 py-2 border border-gray-300
rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
placeholder="e.g., 10"
/>
</div>
<div>
<label className="block text-sm font-medium text-
gray-700 mb-2">
Price per pack (local currency)
</label>
<input
type="number"
step="0.01"
value={formData.pricePerPack}
onChange={(e) => handleInputChange('pricePerPack',
e.target.value)}
className="w-full px-4 py-2 border border-gray-300
rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
placeholder="e.g., 12.50"
/>
</div>
<div>
<label className="block text-sm font-medium text-
gray-700 mb-2">
Quit date (or planned quit date)
</label>
<input
type="date"
value={formData.quitDate}
onChange={(e) => handleInputChange('quitDate',
e.target.value)}
className="w-full px-4 py-2 border border-gray-300
rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
/>
</div>
</div>
<div className="space-y-4">
<h3 className="text-lg font-semibold text-gray-
700">Personal Information</h3>
<div>
<label className="block text-sm font-medium text-
gray-700 mb-2">
Age
</label>
<input
type="number"
value={formData.age}
onChange={(e) => handleInputChange('age',
e.target.value)}
className="w-full px-4 py-2 border border-gray-300
rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
placeholder="e.g., 35"
/>
</div>
<div>
<label className="block text-sm font-medium text-
gray-700 mb-2">
Gender
</label>
<select
value={formData.gender}
onChange={(e) => handleInputChange('gender',
e.target.value)}
className="w-full px-4 py-2 border border-gray-300
rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
>
<option value="">Select gender</option>
<option value="male">Male</option>
<option value="female">Female</option>
<option value="other">Other</option>
</select>
</div>
<div>
<label className="block text-sm font-medium text-
gray-700 mb-2">
Previous quit attempts
</label>
<input
type="number"
value={formData.previousAttempts}
onChange={(e) =>
handleInputChange('previousAttempts', e.target.value)}
className="w-full px-4 py-2 border border-gray-300
rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
placeholder="0"
/>
</div>
<div>
<label className="block text-sm font-medium text-
gray-700 mb-2">
Current stress level (1-5)
</label>
<input
type="range"
min="1"
max="5"
value={formData.stressLevel}
onChange={(e) => handleInputChange('stressLevel',
e.target.value)}
className="w-full"
/>
<div className="flex justify-between text-xs text-
gray-500 mt-1">
<span>Low</span>
<span>High</span>
</div>
</div>
</div>
</div>
{/* Health Conditions */}
<div>
<h3 className="text-lg font-semibold text-gray-700 mb-
4">Health Conditions</h3>
<div className="grid grid-cols-2 md:grid-cols-3 gap-3">
{healthConditions.map(condition => (
<label key={condition} className="flex items-center
space-x-2 cursor-pointer">
<input
type="checkbox"
checked={formData.healthConditions.includes(condition)}
onChange={() =>
handleArrayToggle('healthConditions', condition)}
className="rounded border-gray-300 text-blue-600
focus:ring-blue-500"
/>
<span className="text-sm text-gray-
700">{condition}</span>
</label>
))}
</div>
</div>
{/* Smoking Triggers */}
<div>
<h3 className="text-lg font-semibold text-gray-700 mb-
4">Primary Smoking Triggers</h3>
<div className="grid grid-cols-2 md:grid-cols-3 gap-3">
{triggers.map(trigger => (
<label key={trigger} className="flex items-center
space-x-2 cursor-pointer">
<input
type="checkbox"
checked={formData.smokingTriggers.includes(trigger)}
onChange={() =>
handleArrayToggle('smokingTriggers', trigger)}
className="rounded border-gray-300 text-blue-600
focus:ring-blue-500"
/>
<span className="text-sm text-gray-
700">{trigger}</span>
</label>
))}
</div>
</div>
<button
onClick={calculateResults}
className="w-full bg-gradient-to-r from-blue-600 to-
green-600 text-white py-4 px-6 rounded-lg font-semibold text-lg
hover:from-blue-700 hover:to-green-700 transition-all duration-200
transform hover:scale-105"
>
Calculate My Smoking Impact & Savings
</button>
</div>
)}
{activeTab === 'eq-assessment' && (
<div className="space-y-8">
<div className="text-center mb-8">
<Brain className="w-16 h-16 mx-auto text-purple-600 mb-4"
/>
<h2 className="text-2xl font-bold text-gray-
800">Emotional Quotient Assessment</h2>
<p className="text-gray-600 mt-2">
Assess your emotional readiness to quit smoking
successfully
</p>
</div>
{eqQuestions.map((question, index) => (
<div key={question.id} className="bg-gray-50 p-6 rounded-
lg">
<h3 className="text-lg font-medium text-gray-800 mb-4">
{index + 1}. {question.question}
</h3>
<div className="space-y-2">
{question.options.map((option, optionIndex) => (
<label key={optionIndex} className="flex items-
center space-x-3 cursor-pointer">
<input
type="radio"
name={question.id}
value={optionIndex}
checked={eqAnswers[question.id] ===
optionIndex}
onChange={() => handleEQAnswer(question.id,
optionIndex)}
className="text-purple-600 focus:ring-purple-
500"
/>
<span className="text-gray-700">{option}</span>
</label>
))}
</div>
</div>
))}
{eqScore && (
<div className="bg-purple-50 border border-purple-200
rounded-lg p-6">
<h3 className="text-xl font-bold text-purple-800 mb-3">
Your EQ Score: {eqScore.score}/100
</h3>
<p className="text-purple-700 mb-4 font-medium">
{eqScore.interpretation}
</p>
<div>
<h4 className="font-semibold text-purple-800 mb-
2">Personalized Recommendations:</h4>
<ul className="space-y-2">
{eqScore.recommendations.map((rec, index) => (
<li key={index} className="text-purple-700 flex
items-start">
<CheckCircle className="w-4 h-4 mr-2 mt-0.5
text-purple-600 flex-shrink-0" />
{rec}
</li>
))}
</ul>
</div>
</div>
)}
</div>
)}
{activeTab === 'results' && !results && (
<div className="text-center py-12">
<Calculator className="w-16 h-16 mx-auto text-gray-400 mb-
4" />
<h3 className="text-xl font-semibold text-gray-600 mb-
2">Complete the Calculator First</h3>
<p className="text-gray-500">Fill out your smoking profile
to see detailed results and insights.</p>
</div>
)}
{activeTab === 'insights' && results && eqScore && (
<div className="space-y-8">
<h2 className="text-2xl font-bold text-gray-800 text-center
mb-8">Personalized Quit Strategy</h2>
{/* Success Probability */}
<div className="bg-gradient-to-r from-green-100 to-blue-100
border border-green-200 rounded-lg p-6">
<h3 className="text-xl font-bold text-gray-800 mb-4 flex
items-center">
<Target className="w-6 h-6 mr-2" />
Success Probability Analysis
</h3>
<div className="grid md:grid-cols-2 gap-6">
<div>
<h4 className="font-semibold text-gray-700 mb-
2">Positive Factors:</h4>
<ul className="space-y-1">
{eqScore.score > 60 && (
<li className="text-green-700 flex items-center">
<CheckCircle className="w-4 h-4 mr-2" />
Strong emotional readiness
</li>
)}
{formData.supportSystem >= 4 && (
<li className="text-green-700 flex items-center">
<CheckCircle className="w-4 h-4 mr-2" />
Good support system
</li>
)}
{parseFloat(formData.age) < 40 && (
<li className="text-green-700 flex items-center">
<CheckCircle className="w-4 h-4 mr-2" />
Younger age advantage
</li>
)}
{formData.smokingTriggers.length <= 3 && (
<li className="text-green-700 flex items-center">
<CheckCircle className="w-4 h-4 mr-2" />
Limited trigger complexity
</li>
)}
</ul>
</div>
<div>
<h4 className="font-semibold text-gray-700 mb-
2">Areas for Focus:</h4>
<ul className="space-y-1">
{eqScore.score < 60 && (
<li className="text-orange-700 flex items-
center">
<AlertCircle className="w-4 h-4 mr-2" />
Build emotional coping skills
</li>
)}
{formData.previousAttempts > 2 && (
<li className="text-orange-700 flex items-
center">
<AlertCircle className="w-4 h-4 mr-2" />
Learn from previous attempts
</li>
)}
{formData.stressLevel >= 4 && (
<li className="text-orange-700 flex items-
center">
<AlertCircle className="w-4 h-4 mr-2" />
Prioritize stress management
</li>
)}
{formData.smokingTriggers.length > 5 && (
<li className="text-orange-700 flex items-
center">
<AlertCircle className="w-4 h-4 mr-2" />
Address multiple triggers systematically
</li>
)}
</ul>
</div>
</div>
</div>
{/* Personalized Action Plan */}
<div className="bg-blue-50 border border-blue-200 rounded-
lg p-6">
<h3 className="text-xl font-bold text-blue-800 mb-4">Your
Personalized 21-Day Action Plan</h3>
<div className="space-y-4">
<div className="bg-white p-4 rounded border-l-4 border-
blue-500">
<h4 className="font-semibold text-blue-800 mb-2">Week
1 Focus: Foundation & Crisis Management</h4>
<ul className="text-blue-700 space-y-1">
<li>• Practice emergency coping techniques every 2
hours</li>
<li>• Address your primary triggers:
{formData.smokingTriggers.slice(0, 3).join(', ')}</li>
<li>• Daily check-ins with support person</li>
{eqScore.score < 60 && <li>• Begin stress
management training immediately</li>}
</ul>
</div>
<div className="bg-white p-4 rounded border-l-4 border-
green-500">
<h4 className="font-semibold text-green-800 mb-
2">Week 2 Focus: Pattern Building</h4>
<ul className="text-green-700 space-y-1">
<li>• Establish new routines for smoking times</li>
<li>• Practice social situations without
smoking</li>
<li>• Build physical activity habits</li>
<li>• Strengthen emotional regulation skills</li>
</ul>
</div>
<div className="bg-white p-4 rounded border-l-4 border-
purple-500">
<h4 className="font-semibold text-purple-800 mb-
2">Week 3 Focus: Long-term Success</h4>
<ul className="text-purple-700 space-y-1">
<li>• Challenge yourself with difficult
situations</li>
<li>• Plan for month 2 and beyond</li>
<li>• Consider supporting others</li>
<li>• Celebrate and reinforce your new
identity</li>
</ul>
</div>
</div>
</div>
{/* Trigger-Specific Strategies */}
{formData.smokingTriggers.length > 0 && (
<div className="bg-gray-50 border border-gray-200
rounded-lg p-6">
<h3 className="text-xl font-bold text-gray-800 mb-
4">Trigger-Specific Coping Strategies</h3>
<div className="grid md:grid-cols-2 gap-4">
{formData.smokingTriggers.map(trigger => (
<div key={trigger} className="bg-white p-4 rounded
border">
<h4 className="font-semibold text-gray-800 mb-
2">{trigger}</h4>
<div className="text-sm text-gray-600">
{trigger === 'Stress' && 'Try: Deep breathing,
5-minute walk, call a friend, progressive muscle relaxation'}
{trigger === 'Social Situations' && 'Try: Bring
gum, hold a drink, practice polite refusals, find non-smoking friends'}
{trigger === 'Alcohol' && 'Try: Limit alcohol
initially, bring a supporter, practice saying no, have an exit plan'}
{trigger === 'Coffee/Meals' && 'Try: Brush
teeth immediately, change seating, drink water, use toothpick'}
{trigger === 'Driving' && 'Try: Remove
lighters, play music, chew gum, take different routes'}
{trigger === 'Work Breaks' && 'Try: Take
walking breaks, call friends, do desk exercises, eat healthy snacks'}
{trigger === 'Boredom' && 'Try: Plan
activities, learn new skills, exercise, call friends, read'}
{trigger === 'Anger' && 'Try: Count to 10,
physical exercise, journal writing, talk to someone'}
{trigger === 'Celebration' && 'Try: Plan non-
smoking celebrations, focus on health victories, reward yourself
differently'}
</div>
</div>
))}
</div>
</div>
)}
</div>
)}
{activeTab === 'insights' && (!results || !eqScore) && (
<div className="text-center py-12">
<Target className="w-16 h-16 mx-auto text-gray-400 mb-4" />
<h3 className="text-xl font-semibold text-gray-600 mb-
2">Complete Assessment First</h3>
<p className="text-gray-500">Fill out the calculator and EQ
assessment to receive personalized insights.</p>
</div>
)}
</div>
{/* SWOT Analysis Feature */}
{results && eqScore && (
<div className="bg-gray-50 border-t border-gray-200 p-8">
<h3 className="text-2xl font-bold text-gray-800 mb-6 text-
center">Your Personal SWOT Analysis</h3>
<div className="grid md:grid-cols-2 gap-6">
{/* Strengths */}
<div className="bg-green-50 border border-green-200
rounded-lg p-6">
<h4 className="text-lg font-bold text-green-800 mb-3 flex
items-center">
<CheckCircle className="w-5 h-5 mr-2" />
Strengths
</h4>
<ul className="text-green-700 space-y-1 text-sm">
{eqScore.score > 70 && <li>• High emotional
intelligence</li>}
{formData.previousAttempts > 0 && <li>• Learning from
previous experiences</li>}
{formData.supportSystem >= 4 && <li>• Strong support
network</li>}
{parseFloat(formData.cigarettesPerDay) < 15 && <li>•
Moderate smoking pattern</li>}
{formData.motivation >= 4 && <li>• High motivation
level</li>}
<li>• Taking proactive steps by using this
calculator</li>
</ul>
</div>
{/* Weaknesses */}
<div className="bg-red-50 border border-red-200 rounded-lg
p-6">
<h4 className="text-lg font-bold text-red-800 mb-3 flex
items-center">
<AlertCircle className="w-5 h-5 mr-2" />
